T1. 百花竞放

这道题目是个签到的题目。其实不用一些非常复杂的算法（像FFTFFTFFT）.

直接用算“贡献”的方式写式子也是一个不错的选择。

对于：

f(i)=∑d∣id2f(i)=\sum_{d|i} d^2 f(i)=d∣i∑​d2

那么每一个iii会给所有的 f(k)f(k)f(k)贡献i2i^2i2的值当且仅当i∣ki|ki∣k.

那么，可以用 接近线性 的时间求出fff函数。

for(int i=1;i<=n;i++) 
for(int j=1;j<=n/i;j++) f[i*j]+=i*i;

当然，也可以用 线性时间 求出ggg函数。

for(int i=1;i<=n;i++) g[i]=i*(i+1)/2;

对于每一个f(i)f(i)f(i)，会给h=i×jh=i \times jh=i×j添加fi×gjf_i \times g_jfi​×gj​的贡献。用类似于推fff的方法进行即可。

for(int i=1;i<=n;i++)
		for(int j=1;j<=n/i;j++)
			h[i*j]=(h[i*j]+f[i]*g[j])%MOD;

最后求 ∑i=1nhi\sum_{i=1}^n h_i∑i=1n​hi​也可以线性解决。

T2.简单方差

本题本来出的时候没有DDD函数，被某大佬吐槽 “O(n+m)O(n+m)O(n+m)”解决，因此添加了DDD函数。

不过后来一想，gcdgcdgcd也可以用类似RMQRMQRMQ的方法。因为这个运算可以重复计算啊！也就是说本题线段树可以不用开的！

我们一个个解决询问，可以用很多个数据结构。

fff函数和ggg函数，直接用 前缀和 就可以解决。

注意：∑i=lr=abs(fr−fl−1)\sum_{i=l}^r = abs(f_r-f_{l-1})∑i=lr​=abs(fr​−fl−1​)，因为在模意义下可能出现fr<fl−1f_r<f_{l-1}fr​<fl−1​的情况。同样ggg也是，需要处理。（一开始做stdstdstd结果答案出负数，找了很久）

hhh函数，我们追求最高的效率，用RMQRMQRMQ解决该问题。

AAA函数，仍然用前缀和。

BBB函数推一下式子：

∑i=lr∑j=lrai×aj\sum_{i=l}^r \sum_{j=l}^r a_i \times a_j i=l∑r​j=l∑r​ai​×aj​

=(∑i=1rai)×(∑j=lraj)=(\sum_{i=1}^r a_i) \times (\sum_{j=l}^r a_j) =(i=1∑r​ai​)×(j=l∑r​aj​)

=(∑i=lrai)2=(\sum_{i=l}^r a_i)^2 =(i=l∑r​ai​)2

直接区间和的平方就可以了。（似乎不用处理负数，因为(−a)2=a2(-a)^2=a^2(−a)2=a2）

对于DDD函数，线段树维护。

对于CCC函数，你会发现这就是方差。

令x=∑i=lrr−l+1x=\frac{\sum_{i=l}^r}{r-l+1}x=r−l+1∑i=lr​​，则：

C(l,r)×(r−l+1)=∑i=lr(ai−x)2C(l,r) \times (r-l+1)=\sum_{i=l}^r (a_i-x)^2 C(l,r)×(r−l+1)=i=l∑r​(ai​−x)2

=∑i=lrai2+x2−2×ai×x=\sum_{i=l}^r a_i^2 + x^2 - 2 \times a_i \times x =i=l∑r​ai2​+x2−2×ai​×x

这是你会发现，全部可以用前面处理的前缀和O(1)O(1)O(1)回答！

#include<bits/stdc++.h>
using namespace std;

#define L (i<<1)
#define R (i<<1)+1

typedef long long ll;

inline int read(){char ch=getchar();int f=1;while(ch<'0' || ch>'9') {if(ch=='-') f=-f; ch=getchar();}
	int x=0;while(ch>='0' && ch<='9') x=x*10+ch-'0',ch=getchar();return x*f;}

const int N=5e5+1;
const int MOD=998244353;

struct data{
	int l,r;
	int gcdi;
};
data t[4*N];
int a[N],n,m;
ll s1[N],s2[N],s3[N];
int f[18][N];

inline int gcd(int n,int m) {return m==0?n:gcd(m,n%m);}

inline void update(int i) {
	t[i].gcdi=gcd(t[L].gcdi,t[R].gcdi);
	t[i].maxi=max(t[L].maxi,t[R].maxi);
}

inline void build_tree(int i,int l,int r) {
	t[i].l=l; t[i].r=r;
	if(l==r) {t[i].maxi=a[l];t[i].gcdi=a[l];return;}
	int mid=(l+r)>>1;
	build_tree(L,l,mid);
	build_tree(R,mid+1,r);
	update(i); 
}

inline int querygcd(int i,int l,int r) {
	if(l<=t[i].l && t[i].r<=r) return t[i].gcdi;
	int mid=(t[i].l+t[i].r)>>1;
	if(l<=mid && r>mid) return gcd(querygcd(L,l,r),querygcd(R,l,r));
	if(l<=mid) return querygcd(L,l,r);
	return querygcd(R,l,r);
}

inline int solve(int x) {
	return (x<0)?(x+MOD):x;
}

int main(){
	n=read(),m=read();
	for(int i=1;i<=n;i++) a[i]=read();
	for(int i=1;i<=n;i++) s1[i]=(s1[i-1]+a[i])%MOD,s2[i]=(s2[i-1]+a[i]*a[i])%MOD,
	s3[i]=(s3[i-1]+a[i-1]*a[i])%MOD;
	build_tree(1,1,n);
	for(int i=1;i<=n;i++) f[0][i]=a[i];
	for(int j=1;j<17;j++)
	for(int i=1;i<=n;i++) f[j][i]=max(f[j-1][i],f[j-1][i+(1<<(j-1))]);
	while(m--) {
		char c; cin>>c;
		int l=read(),r=read();
		if(l>r) swap(l,r);
		if(c=='f') printf("%lld\n",solve(s1[r]-s1[l-1]));
		if(c=='g') printf("%lld\n",solve(s2[r]-s2[l-1]));
		if(c=='h') {
			int s=log(r-l+1)/log(2);
			printf("%d\n",max(f[s][l],f[s][r-(1<<s)+1]));
		}
		if(c=='A') printf("%lld\n",solve(s3[r]-s3[l]));
		if(c=='B') printf("%lld\n",solve(s1[r]-s1[l-1])*solve(s1[r]-s1[l-1]));
		if(c=='C') {
			double avg=solve(s1[r]-s1[l-1])*1.0/(r-l+1);
			double x=solve(s2[r]-s2[l-1]); double k=solve(s1[r]-s1[l-1])*solve(s1[r]-s1[l-1]);
			printf("%.2lf\n",solve(s2[r]-s2[l-1])*1.0/(r-l+1)-k/(r-l+1)/(r-l+1));
		}
		if(c=='D') printf("%d\n",querygcd(1,l,r));
	}
	return 0;
}

Case 3.梅兰竹菊

这题的难度非常的低。

首先可以证明，答案≤1\leq 1≤1.

方法：枚举一个数，然后用xxx除掉它，再分解成两个数的乘积验证，旁边开个哈希就完事了。

#include<bits/stdc++.h>
using namespace std;

typedef long long ll; 

inline ll read_ll(){char ch=getchar();int f=1;while(ch<'0' || ch>'9') {if(ch=='-') f=-f; ch=getchar();}
	ll x=0;while(ch>='0' && ch<='9') x=x*10+ch-'0',ch=getchar();return x*f;}

inline int read_int(){char ch=getchar();int f=1;while(ch<'0' || ch>'9') {if(ch=='-') f=-f; ch=getchar();}
	int x=0;while(ch>='0' && ch<='9') x=x*10+ch-'0',ch=getchar();return x*f;}

const int N=1e6+1;

int n,k,a[N];
ll x; int ans=0,maxi;
bool d[N];

inline void check(ll q,int k,int p) {
	if((ll)(maxi*maxi)<q) return;
	for(int i=1;(ll)(i*i)<q && i<=maxi;i++)
		if(q%i==0 && (ll)(q/i)<=maxi && d[i] && d[q/i] && i+q/i==k && i!=p && q/i!=p) {
			putchar('1');
			exit(0);
		}
} 

int main(){
	n=read_int(),k=read_int(),x=read_ll();
	int l=0;
	for(int i=1,y;i<=n;i++) {y=read_int();if(y && y<=k-2 && (ll)(x%y)==0 && !d[y]) 
	a[++a[0]]=y,d[y]=1,maxi=max(maxi,y);}
	if(maxi*3+2<=k) {putchar('0');return 0;} 
	n=a[0];
	for(int i=1;i<=n;i++) check(x/a[i],k-a[i],a[i]);
	putchar('0');
	return 0;
}

Case 4:树干之谜

乍一眼看上去很难，还有累乘？其实就是 小学奥数题 。

比方说n=3n=3n=3，m=3m=3m=3时：

原式

=1×2×3+2×3×4+3×4×5=1 \times 2 \times 3 + 2 \times 3 \times 4 +3 \times 4 \times 5=1×2×3+2×3×4+3×4×5

=14×(1×2×3×4−0×1×2×3+2×3×4×5−1×2×3×4+3×4×5×6−2×3×4×5)=\frac{1}{4} \times (1 \times 2 \times 3 \times 4 - 0 \times 1 \times 2 \times 3 + 2 \times 3 \times 4 \times 5 - 1 \times 2 \times 3 \times 4 + 3 \times 4 \times 5 \times 6 - 2 \times 3 \times 4 \times 5)=41​×(1×2×3×4−0×1×2×3+2×3×4×5−1×2×3×4+3×4×5×6−2×3×4×5)

=14×3×4×5×6=\frac{1}{4} \times 3 \times 4 \times 5 \times 6=41​×3×4×5×6

读者可以自行再举几个例子证明。

你会发现：答案就是：

1m+1×n×(n+1)×...×(n+m)\frac{1}{m+1} \times n \times (n+1) \times ... \times (n+m) m+11​×n×(n+1)×...×(n+m)

答案很显然：预处理阶乘，快速幂（标程不知怎么就写了个exgcdexgcdexgcd）逆元。

#include<bits/stdc++.h>
using namespace std;

typedef long long ll;

inline ll read(){char ch=getchar();int f=1;while(ch<'0' || ch>'9') {if(ch=='-') f=-f; ch=getchar();}
	ll x=0;while(ch>='0' && ch<='9') x=x*10+ch-'0',ch=getchar();return x*f;}

const ll MOD=1e9+7;
ll n,m;
ll jc[20000001];

inline ll pow(ll x,ll y) {
	ll ans=1;
	while(y>0) {
		if(y&1) ans=(ans*x)%MOD;
		y>>=1; x=(x*x)%MOD;
	} return ans;
}

ll x,y;

inline void exgcd(ll a,ll b) {
	if(!b) {
		x=1,y=0;
		return;
	}
	exgcd(b,a%b);
	ll t=x; x=y; y=t-a/b*y;
} 

int main(){
	n=read(),m=read();
	jc[1]=1;
	for(int i=2;i<=n+m;i++) jc[i]=(ll)(jc[i-1]*i)%MOD,jc[i]=(jc[i]<0)?(jc[i]+MOD):(jc[i]);
	if(n==1) {printf("%lld\n",jc[m]);return 0;}
	ll a=jc[n+m],b=(jc[n-1]*(m+1))%MOD; if(b<0) b+=MOD; 
	exgcd(b,MOD);
	x=(x%MOD+MOD)%MOD;
	printf("%lld\n",a*(ll)x%MOD);
	return 0;
}
